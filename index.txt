<script>
    // Register Service Worker
    if ('serviceWorker' in navigator) {
        navigator.serviceWorker.register('sw.js');
    }

    function loadStats() {
        const name = localStorage.getItem('remi_last_player') || "New Driver";
        const time = localStorage.getItem('remi_l1_time') || "--:--";
        const miles = localStorage.getItem('remi_l1_miles') || "0.0";
        
        document.getElementById('stat-name').innerText = name;
        document.getElementById('stat-time').innerText = time;
        document.getElementById('stat-miles').innerText = miles;
    }

    function clearStats() {
        if(confirm("Clear all game progress?")) {
            localStorage.clear();
            location.reload();
        }
    }

    window.onload = loadStats;
</script>
